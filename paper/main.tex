\begin{figure}
    \includegraphics{k5/ke_oinfo.pdf}
    \caption{$k_e^*$ correlates negatively with O-information 
    ($\rho=-0.293, \; p < 0.05$) but shows no obvious functional relationship.
    This lack of a clear relationship suggests that O-information is not a useful
    measure of $k_e$.}
    \label{fig:ke-oinfo}
\end{figure}

\begin{figure}
    \includegraphics{k5/directed_cana.pdf}
    \caption{Directed information measures transfer entropy and active 
    information storage are significantly but weakly negatively correlated with
    $k_e$ ($\rho = -0.127, \; p<0.05$). As with O-information, a lack of obvious
    functional relationship make these weak measures of redundancy.}
    \label{fig:r-te}
\end{figure}

\begin{figure}
    \includegraphics{k5/corr_mat.pdf}
    \caption{Correlation matrix for interpretable partial information atoms and
    redundancy measures. Notable correlations are those with $k_e$, particularly
    the positive correlation with the pure redundancy $\{0\}\{1\}\{2\}\{3\}\{4\}$
    ($\rho = 0.592, \; p<0.05$) and pure synergy $\{0:1:2:3:4\}$ 
    ($\rho=0.580, \; p<0.05$).}
    \label{fig:corr-mat}
\end{figure}
